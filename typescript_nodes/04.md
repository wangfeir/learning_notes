#### 4、接口中定义只读参数 
```javascript
interface OnlyArrey{
  readonly x:string //定义只读类型就是在参数的前面添加readonly
  readonly y:string
}
let myObj:OnlyArrey = {
  x:'1111',
  y:'2222',
}
myObj.x = 2222 // 接口中把这个值定义为只读类型是不可更改的，这里会报错
console.log(myObj)
//定义只读数组
let onlyArray:ReadonlyArray<number> = [1,2,3,34,4,5,7]
onlyArray[2] = 2 // 这是会报错的

```
